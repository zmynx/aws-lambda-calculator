28def unit_conversion_requests( 

29 number_of_requests: int, request_unit: str, steps: list[str] 

30) -> int: 

31 """ 

32 @brief Convert number of requests based on the unit provided. Assuming 730 hours in a month (30 days). Assuming 24 hours in a day. 

33 @param number_of_requests: The number of requests to convert. 

34 @param request_unit: per second, per minute, per hour, per day, per month, million per month. 

35 @return: The number of requests per month. 

36 """ 

37 match request_unit: 

38 case "per second": 

39 msg = f"Number of requests: {number_of_requests:,} per second * (60 seconds in a minute * 60 minutes in an hour * 730 hours in a month) = {number_of_requests * (60 * 60 * 730):,} per month" 

40 logger.debug(msg) 

41 steps.append(msg) 

42 return int(number_of_requests * (60 * 60 * 730)) 

43 case "per minute": 

44 msg = f"Number of requests: {number_of_requests:,} per minute * (60 minutes in an hour * 730 hours in a month) = {int(number_of_requests * (60 * 730)):,} per month" 

45 logger.debug(msg) 

46 steps.append(msg) 

47 return int(number_of_requests * (60 * 730)) 

48 case "per hour": 

49 msg = f"Number of requests: {number_of_requests:,} per hour * (730 hours in a month) = {int(number_of_requests * 730):,} per month" 

50 logger.debug(msg) 

51 steps.append(msg) 

52 return int(number_of_requests * (730)) 

53 case "per day": 

54 msg = f"Number of requests: {number_of_requests:,} per day * (730 hours in a month / 24 hours in a day) = {int(number_of_requests * (730 / 24)):,} per month" 

55 logger.debug(msg) 

56 steps.append(msg) 

57 return int(number_of_requests * (730 / 24)) 

58 case "per month": 

59 msg = f"Number of requests: {int(number_of_requests):,} per month" 

60 logger.debug(msg) 

61 steps.append(msg) 

62 return int(number_of_requests)

64 msg = f"Number of requests: {number_of_requests} million per month * 1,000,000 multiplier = {int(number_of_requests * 1000000):,} per month" 

65 logger.debug(msg)

68 case _:

163def calc_monthly_compute_charges( 

164 requests_per_month: int, 

165 duration_of_each_request_in_ms: int, 

166 memory_in_gb: float, 

167 tier_cost_factor: dict[str, float], 

168 include_free_tier: bool, 

169 steps: list[str], 

170) -> tuple[float, float, float]: 

171 """ 

172 @brief Calculate the monthly compute charges based on requests per month, duration of each request in ms, and memory in GB. 

173 @param requests_per_month: The number of requests per month. 

174 @param duration_of_each_request_in_ms: The duration of each request in milliseconds. 

175 @param memory_in_gb: The amount of memory allocated in GB. 

176 @return: The monthly compute charges. 

177 """ 

178 total_compute_sec = requests_per_month * (duration_of_each_request_in_ms * 0.001) 

179 msg = f"{requests_per_month} requests x {duration_of_each_request_in_ms} ms x 0.001 ms to sec conversion factor = {total_compute_sec} total compute (seconds)" 

180 logger.debug(msg) 

181 steps.append(msg) 

182 

183 total_compute_gb_sec = memory_in_gb * total_compute_sec 

184 msg = f"{memory_in_gb} GB x {total_compute_sec:,} seconds = {total_compute_gb_sec:,.2f} total compute (GB-s)" 

185 logger.debug(msg) 

186 steps.append(msg) 

187 

188 ## Apply free tier for compute if enabled 

189 billable_compute_gb_sec = total_compute_gb_sec 

190 if include_free_tier: 

191 free_compute_gb_sec = 400_000 # 400,000 GB-seconds per month 

192 billable_compute_gb_sec = max(0.0, total_compute_gb_sec - free_compute_gb_sec) 

193 msg = f"{total_compute_gb_sec:,.2f} GB-s - {free_compute_gb_sec:,} free tier GB-s = {billable_compute_gb_sec:,.2f} billable GB-s" 

194 steps.append(msg) 

195 logger.debug(msg) 

196 

197 ## Tiered price for billable compute GB-seconds 

198 msg = f"Tiered price for: {billable_compute_gb_sec:,.2f} GB-s" 

199 logger.debug(msg) 

200 steps.append(msg) 

201 

202 # anything above 15 B GB‑sec 

203 overflow_rate = 0.0000133334 

204 monthly_compute_charges = calculate_tiered_cost( 

205 billable_compute_gb_sec, tier_cost_factor, overflow_rate, steps 

206 ) 

207 return total_compute_gb_sec, monthly_compute_charges, total_compute_sec

254# Flow: 

255# 

256# 1. Detect the correct region 

257# 2. Open the corresponding config file for region from step 1 

258# 3. Extract the cost factores (pricing) from the config file: 

259# 3.1 architecture (x86, arm) 

260# 3.2 memory 

261# 3.3 ephemeral storage 

262# 4. Run unit conversions on the user's input: 

263# 4.1 unit_conversion_requests on @number of requests, @request unit to convert it to requests per month. 

264# 4.2 unit_conversion_memory on @memory, @memory unit to convert it to memory in GB. 

265# 4.3 unit_conversion_ephemeral_storage on @ephemeral storage, @storage unit to convert it to ephemeral storage in GB. 

266# 5. Pricing calculations: 

267# 

268# ## Monthly Compute Charges 

269# 5.1 Calculate total compute seconds based on requests per month and duration of each request in ms. 

270# 5.2 Calculate total compute GB-seconds based on memory in GB and total compute seconds. 

271# 5.3 Calculate tiered price for total compute GB-seconds using the cost factor from the config file. 

272# 5.4 Calculate monthly compute charges based step 5.3. 

273# 

274# ## Monthly Request Charges 

275# 5.5 Calculate monthly request charges based on requests per month and the cost factor from the config file. 

276# 

277# ## Monthly Ephemeral Storage Charges 

278# 5.6 Calculate total storage GB-seconds based on ephemeral storage in GB and total duration seconds. 

279# 

280# 6. Calculate the total monthly cost by summing up the monthly compute charges, monthly request charges, and monthly ephemeral storage charges.
